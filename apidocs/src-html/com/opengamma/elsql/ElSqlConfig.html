042  /**
043   * The descriptive name.
044   */
045  private final String _name;
046
047  /**
048   * Creates an instance.
049   * 
050   * @param name  a descriptive name for the config, not null
051   */
052  public ElSqlConfig(String name) {
053    _name = name;
054  }
055
056  /**
057   * Gets the config name.
058   * 
059   * @return the config name, not null
060   */
061  public final String getName() {
062    return _name;
063  }
064
065  //-------------------------------------------------------------------------
066  /**
067   * Checks if the value contains a wildcard.
068   * <p>
069   * The default implementation matches % and _, using backslash as an escape character.
070   * This matches Postgres and other databases.
071   * 
072   * @param value  the value to check, not null
073   * @return true if the value contains wildcards
074   */
075  public boolean isLikeWildcard(String value) {
076    boolean escape = false;
077    for (int i = 0; i < value.length(); i++) {
078      char ch = value.charAt(i);
079      if (escape) {
080        escape = false;
081      } else {
082        if (ch == '\\') {
083          escape = true;
084        } else if (ch == '%' || ch == '_') {
085          return true;
086        }
087      }
088    }
089    return false;
090  }
091
092  /**
093   * Gets the suffix to add after LIKE, which would typically be an ESCAPE clause.
094   * <p>
095   * The default implementation returns an empty string.
096   * This matches Postgres and other databases.
097   * <p>
098   * The returned SQL must be end in a space if non-empty.
099   * 
100   * @return the suffix to add after LIKE, not null
101   */
102  public String getLikeSuffix() {
103    return "";
104  }
105
106  //-------------------------------------------------------------------------
107  /**
108   * Alters the supplied SQL to add paging, such as OFFSET-FETCH.
109   * <p>
110   * The default implementation calls {@link #getPaging(int, int)}.
111   * <p>
112   * The returned SQL must be end in a space if non-empty.
113   * 
114   * @param selectToPage  the SELECT statement to page, not null
115   * @param offset  the OFFSET amount, zero to start from the beginning
116   * @param fetchLimit  the FETCH/LIMIT amount, zero to fetch all
117   * @return the updated SELECT, not null
118   */
119  public String addPaging(String selectToPage, int offset, int fetchLimit) {
120    return selectToPage + getPaging(offset, fetchLimit);
121  }
122
123  /**
124   * Gets the paging SQL, such as OFFSET-FETCH.
125   * <p>
126   * The default implementation uses 'FETCH FIRST n ROWS ONLY' or
127   * 'OFFSET n ROWS FETCH NEXT n ROWS ONLY'.
128   * This matches Postgres, HSQL and other databases.
129   * <p>
130   * The returned SQL must be end in a space if non-empty.
131   * 
132   * @param offset  the OFFSET amount, zero to start from the beginning
133   * @param fetchLimit  the FETCH/LIMIT amount, zero to fetch all
134   * @return the SQL to use, not null
135   */
136  public String getPaging(int offset, int fetchLimit) {
137    if (fetchLimit == 0 && offset == 0) {
138      return "";
139    }
140    if (fetchLimit == 0) {
141      return "OFFSET " + offset + " ROWS ";
142    }
143    if (offset == 0) {
144      return "FETCH FIRST " + fetchLimit + " ROWS ONLY ";
145    }
146    return "OFFSET " + offset + " ROWS FETCH NEXT " + fetchLimit + " ROWS ONLY ";
147  }
148
149  //-------------------------------------------------------------------------
150  @Override
151  public String toString() {
152    return "ElSqlConfig[" + _name + "]";
153  }

193  /**
194   * Class for MySQL.
195   */
196  private static class MySqlElSqlConfig extends ElSqlConfig {
197    public MySqlElSqlConfig() {
198      super("MySql");
199    }
200    @Override
201    public String getPaging(int offset, int fetchLimit) {
202      if (fetchLimit == 0 && offset == 0) {
203        return "";
204      }
205      if (fetchLimit == 0) {
206        return "OFFSET " + offset + " ";
207      }
208      if (offset == 0) {
209        return "LIMIT " + fetchLimit + " ";
210      }
211      return "LIMIT " + fetchLimit + " OFFSET " + offset + " ";
212    }
213  }
214
215  //-------------------------------------------------------------------------
216  /**
217   * Class for SQL server 2008.
218   */
219  private static class SqlServer2008ElSqlConfig extends ElSqlConfig {
220    public SqlServer2008ElSqlConfig() {
221      super("SqlServer2008");
222    }
223    @Override
224    public String addPaging(String selectToPage, int offset, int fetchLimit) {
225      if (fetchLimit == 0 && offset == 0) {
226        // SQL Server needs a SELECT TOP with ORDER BY in an inner query, otherwise it complains
227        return selectToPage.replaceFirst("SELECT ", "SELECT TOP " + Integer.MAX_VALUE + " ");
228      }
229      int start = offset + 1;
230      int end = offset + fetchLimit;
231      String columns = selectToPage.substring(selectToPage.indexOf("SELECT ") + 7, selectToPage.indexOf(" FROM "));
232      String from = selectToPage.substring(selectToPage.indexOf(" FROM ") + 6, selectToPage.indexOf(" ORDER BY "));
233      String order = selectToPage.substring(selectToPage.indexOf(" ORDER BY ") + 10);
234      String inner = "SELECT " + columns + ", ROW_NUMBER() OVER (ORDER BY " + order.trim() + ") AS ROW_NUM FROM " + from;
235      return "SELECT * FROM (" + inner + ") AS ROW_TABLE WHERE ROW_NUM >= " + start + " AND ROW_NUM <= " + end;
236    }
237    @Override
238    public String getPaging(int offset, int fetchLimit) {
239      throw new UnsupportedOperationException();
240    }
241  }
